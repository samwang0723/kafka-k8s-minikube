This chart installs a {{ default 3 .Values.replicas }} nodes Kafka Brokers.

To connect from a client pod:

1. Deploy a kafka client pod with configuration:

    apiVersion: v1
    kind: Pod
    metadata:
      name: kafka-client
      namespace: {{ .Release.Namespace }}
    spec:
      containers:
      - name: kafka-client
        image: {{ .Values.images.name }}:{{ .Values.images.tag }}
        command:
          - sh
          - -c
          - "exec tail -f /dev/null"

2. Log into the Pod

  kubectl exec -it kafka-client -- /bin/bash

3. Explore with kafka commands:

  # Delete the topic if it exists
  kafka-topics --zookeeper {{ template "kafka.zookeeper.service-name" . }} --topic {{ template "kafka.fullname" . }}-topic --delete --if-exists

  # Create the topic
  kafka-topics --zookeeper {{ template "kafka.zookeeper.service-name" . }} --topic {{ template "kafka.fullname" . }}-topic --create --partitions 1 --replication-factor 1 --if-not-exists

  # Create a message
  MESSAGE="`date -u`"

  # Produce a test message to the topic
  echo "$MESSAGE" | kafka-console-producer --broker-list {{ template "kafka.fullname" . }}:9092 --topic {{ template "kafka.fullname" . }}-topic && \

  # Consume a test message from the topic
  kafka-console-consumer --bootstrap-server {{ template "kafka.fullname" . }}-headless:9092 --topic {{ template "kafka.fullname" . }}-topic --from-beginning --timeout-ms 2000 | grep "$MESSAGE"
